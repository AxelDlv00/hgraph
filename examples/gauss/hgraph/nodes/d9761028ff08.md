theorem isEven_add {a b : Nat} (ha : IsEven a) (hb : IsEven b) :
    IsEven (a + b) := by
  obtain ⟨i, rfl⟩ := ha
  obtain ⟨j, rfl⟩ := hb
  exact ⟨i + j, by ring⟩
